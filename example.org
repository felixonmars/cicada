* int

  #+begin_src scheme
  (~ square (: int int -> int)
     dup mul end)

  (~ factorial (: int -> int)
     (if dup (# 1) lteq? -> drop (# 1) end)
     dup (# 1) sub factorial mul end)

  (~ t (: ->)
     (# 123 -321) dup square . . .
     (# 0) factorial .
     (# 1) factorial .
     (# 2) factorial .
     (# 3) factorial .
     (# 4) factorial .
     end)
  t
  #+end_src
